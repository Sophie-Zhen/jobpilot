% Sections are emitted in the order given by section_order so the layout can be
% tuned per candidate profile (e.g. projects above experience for career changers).
\BLOCK{for _section in section_order}

\BLOCK{if _section == 'summary'}
\BLOCK{if summary}
\section*{Professional Summary}
\VAR{summary}
\BLOCK{endif}

\BLOCK{elif _section == 'experience'}
\BLOCK{if experience}
\section*{Experience}
\BLOCK{for exp in experience}
\textbf{\VAR{exp.title}} \hfill \VAR{exp.dates}\\
\VAR{exp.company} \hfill \VAR{exp.location}
\begin{itemize}[leftmargin=1.5em, topsep=1pt, itemsep=1pt, parsep=0pt]
\BLOCK{for bullet in exp.bullets}
    \item \VAR{bullet}
\BLOCK{endfor}
\end{itemize}
\BLOCK{if exp.tech}
\textit{Technologies:} \VAR{exp.tech | join(', ')}
\BLOCK{endif}
\vspace{2pt}
\BLOCK{endfor}
\BLOCK{endif}

\BLOCK{elif _section == 'education'}
\BLOCK{if education}
\section*{Education}
\BLOCK{for edu in education}
\textbf{\VAR{edu.degree}} \hfill \VAR{edu.dates}\\
\VAR{edu.institution} \hfill \VAR{edu.location}
\BLOCK{if edu.details}
\begin{itemize}[leftmargin=1.5em, topsep=1pt, itemsep=0pt, parsep=0pt]
\BLOCK{for detail in edu.details}
    \item \VAR{detail}
\BLOCK{endfor}
\end{itemize}
\BLOCK{endif}
\vspace{2pt}
\BLOCK{endfor}
\BLOCK{endif}

\BLOCK{elif _section == 'projects'}
\BLOCK{if projects}
\section*{Projects}
\BLOCK{for proj in projects}
\textbf{\VAR{proj.title}}
\BLOCK{if proj.tech} \quad $\vert$ \quad \textit{\VAR{proj.tech}}\BLOCK{endif}
\BLOCK{if proj.dates} \hfill \VAR{proj.dates}\BLOCK{endif}
\begin{itemize}[leftmargin=1.5em, topsep=1pt, itemsep=1pt, parsep=0pt]
\BLOCK{for bullet in proj.bullets}
    \item \VAR{bullet}
\BLOCK{endfor}
\end{itemize}
\vspace{2pt}
\BLOCK{endfor}
\BLOCK{endif}

\BLOCK{elif _section == 'skills'}
\BLOCK{if skills}
\section*{Skills}
\BLOCK{if skills is mapping}
\BLOCK{for category, items in skills.items()}
\textbf{\VAR{category}:} \VAR{items | join(', ')}\\
\BLOCK{endfor}
\BLOCK{else}
\VAR{skills | join(', ')}
\BLOCK{endif}
\BLOCK{endif}

\BLOCK{elif _section == 'awards'}
\BLOCK{if awards}
\section*{Awards}
\begin{itemize}[leftmargin=1.5em, topsep=1pt, itemsep=1pt, parsep=0pt]
\BLOCK{for award in awards}
    \item \VAR{award}
\BLOCK{endfor}
\end{itemize}
\BLOCK{endif}

\BLOCK{endif}

\BLOCK{endfor}
